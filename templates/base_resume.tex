% A4 paper size by default, use 'letterpaper' for US letter
\documentclass[<<font_size>>, <<paper_size>>]{../src/awesome-cv}

% Configure page margins with geometry
\geometry{left=<<margins.left>>, top=<<margins.top>>, right=<<margins.right>>, bottom=<<margins.bottom>>, footskip=<<margins.footskip>>}

% Improve line breaking and prevent overfull boxes
\sloppy
\setlength{\emergencystretch}{3em}

% Custom spacing adjustments
\renewcommand{\acvSectionTopSkip}{-2mm}

% Prevent duplicate contact info in PDF metadata and text layer
\makeatletter
\AtBeginDocument{
  \hypersetup{
    pdftitle={ <<name.first>> <<name.last>> - <<position.resume>> },
    pdfauthor={ <<name.first>> <<name.last>> },
    pdfsubject={Resume},
    pdfkeywords={Software Engineer, Full Stack Developer, TypeScript, React, Node.js, AWS, GCP, Docker, CI/CD},
    pdfcreator={XeLaTeX with Awesome-CV}
  }
}
\makeatother
\renewcommand{\acvSectionContentTopSkip}{2mm}

% Custom spacing for cvitems (bullet points)
\renewenvironment{cvitems}{%
  \vspace{-8pt}%
  \begin{justify}
  \begin{itemize}[leftmargin=2ex, topsep=2pt, itemsep=1pt, partopsep=0pt, parsep=0pt]%
    \setlength{\parskip}{0pt}%
    \renewcommand{\labelitemi}{\bullet}%
}{%
  \end{itemize}
  \end{justify}
  \vspace{8pt}%
}

% Color the first word of section titles
\makeatletter
\renewcommand{\@sectioncolor}[1]{%
  \ifbool{acvSectionColorHighlight}{%
    \@tempcnta=1\relax
    \@tfor\@tempa:=#1\do{%
      \ifnum\@tempcnta=1\relax
        \color{awesome}\@tempa%
      \else
        \ifx\@tempa\@empty\else\ifx\@tempa~ \color{text}\@tempa\expandafter\@gobble\else\color{text}#1\fi\fi%
        \expandafter\@gobbletwo
      \fi
      \global\advance\@tempcnta\@ne
    }%
  }{#1}%
}
\makeatother

% Color for highlights
\colorlet{awesome}{<<color>>}

% Set false if you don't want to highlight section with awesome color
\setbool{acvSectionColorHighlight}{<<section_color_highlight>>}

% If you would like to change the social information separator from a pipe (|) to something else
\renewcommand{\acvHeaderSocialSep}{\quad\textbar\quad}

% Reduce spacing between first and last name in header (more reduction for tighter spacing)
\renewcommand{\acvHeaderNameDelim}{\kern-0.15em\space}

\begin{document}

\pdfstringdefDisableCommands{%
  \def\href#1#2{#2}%
  \def\https{https://}%
  \def\http{http://}%
}

% Remove footer completely to prevent duplicate contact info
\renewcommand{\makecvfooter}[3]{} % Empty footer

% Set up ATS-friendly footer with just page number
\fancypagestyle{plain}{%
    \fancyhf{} % Clear all header and footer fields
    \renewcommand{\headrulewidth}{0pt} % Remove header rule if present
    \fancyfoot[C]{\thepage} % Only show page number in the center
}

% Print the header with above personal information
% Give optional argument to change alignment(C: center, L: left, R: right)
\makecvheader[<<header_alignment>>]

% No need for separate footer since we're showing contact info in header
\makecvfooter{}{}{}
