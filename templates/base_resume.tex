% Color the first word of section titles
\makeatletter
\renewcommand{\@sectioncolor}[1]{%
  \ifbool{acvSectionColorHighlight}{%
    \@tempcnta=1\relax
    \@tfor\@tempa:=#1\do{%
      \ifnum\@tempcnta=1\relax
        \color{awesome}\@tempa%
      \else
        \ifx\@tempa\@empty\else\ifx\@tempa~ \color{text}\@tempa\expandafter\@gobble\else\color{text}#1\fi\fi%
        \expandafter\@gobbletwo
      \fi
      \global\advance\@tempcnta\@ne
    }%
  }{#1}%
}
\makeatother

% Parent-first Scrutable™ identity and governed record rail.
\newcommand{\scrutablerecordrail}{%
  \noindent
  \begin{minipage}[t]{0.56\textwidth}
    {\fontsize{13pt}{14pt}\selectfont\bfseries\color{scrutable-ink} <<identity.marks.wordmark>>}\par
    {\fontsize{7pt}{8pt}\selectfont\color{scrutable-body} <<identity.marks.promise>>}
  \end{minipage}%
  \hfill
  \begin{minipage}[t]{0.40\textwidth}
    \raggedleft\ttfamily\fontsize{6.5pt}{8pt}\selectfont\color{scrutable-body}
    <<record.id>> \textbullet{} REV <<record.revision>>\\
    <<record.class>> \textbullet{} <<record.status>>\\
    EFFECTIVE <<record.effective_date>>\quad{\color{scrutable-ink}\rule{5pt}{5pt}}
  \end{minipage}
  \par\vspace{3pt}{\color{scrutable-ink}\hrule}\vspace{<@ if compact_layout @>5<@ else @>7<@ endif @>pt}
}
