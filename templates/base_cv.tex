    {\fontsize{7pt}{8pt}\selectfont\color{scrutable-body} <<identity.marks.promise>>}

  \hfill
  \begin{minipage}[t]{0.40\textwidth}
    \raggedleft\ttfamily\fontsize{6.5pt}{8pt}\selectfont\color{scrutable-body}
    <<record.id>> \textbullet{} REV <<record.revision>>\\
    <<record.class>> \textbullet{} <<record.status>>\\
    EFFECTIVE <<record.effective_date>>\quad{\color{scrutable-ink}\rule{5pt}{5pt}}
  \end{minipage}
  \par\vspace{3pt}{\color{scrutable-ink}\hrule}\vspace{7pt}
}
